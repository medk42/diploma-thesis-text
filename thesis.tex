\ExplSyntaxOn
\cs_new_protected:Npn \jakub_type:n #1
  {
    % KEEP Nx, but do NOT expand #1 (preserve tokens like \rightarrow)
    \tl_set:Nx \l_tmpa_tl { \exp_not:n { #1 } }

    % Break AFTER common separators
    \tl_replace_all:Nnn \l_tmpa_tl { \c_colon_str\c_colon_str } { \c_colon_str\c_colon_str \c{allowbreak} }

    % If user writes "_" literally
    \tl_replace_all:Nnn \l_tmpa_tl { \c_underscore_str } { \c_underscore_str \c{allowbreak} }
    % If user writes "\_" (common in LaTeX code-ish text)
    \tl_replace_all:Nnn \l_tmpa_tl { \_ } { \_ \allowbreak }

    \tl_replace_all:Nnn \l_tmpa_tl { \c_dot_str }    { \c_dot_str \c{allowbreak} }
    \tl_replace_all:Nnn \l_tmpa_tl { \c_slash_str }  { \c_slash_str \c{allowbreak} }
    \tl_replace_all:Nnn \l_tmpa_tl { \c_minus_str }  { \c_minus_str \c{allowbreak} }
    \tl_replace_all:Nnn \l_tmpa_tl { \c_lparen_str } { \c_lparen_str \c{allowbreak} }
    \tl_replace_all:Nnn \l_tmpa_tl { \c_rparen_str } { \c_rparen_str \c{allowbreak} }
    \tl_replace_all:Nnn \l_tmpa_tl { \c_comma_str }  { \c_comma_str \c{allowbreak} }

    % Break BETWEEN lower/digit and Upper (CamelCase)
    \regex_replace_all:nnN
      { ([a-z0-9])([A-Z]) }
      { \1 \c{allowbreak} \2 }
      \l_tmpa_tl

    % IMPORTANT: never end with \allowbreak (fixes dataType_. / dataType_.")
    \regex_replace_once:nnN { \c{allowbreak}\Z } { } \l_tmpa_tl

    \texttt{\tl_use:N \l_tmpa_tl}
  }
